\definecolor{BookForgeMuted}{HTML}{6F685F}

\definecolor{BookForgeRule}{HTML}{D8C9B7}

\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\footnotesize\color{BookForgeMuted}\nouppercase{\leftmark}}

\usepackage{titlesec}
\titleformat{\chapter}[display]
  {\normalfont\sffamily\bfseries\color{BookForgeInk}}
  {\filright\Large\color{BookForgeAccent}\chaptertitlename\ \thechapter}
  {0.7em}
  {\filright\Huge}
  [\vspace{0.6em}{\color{BookForgeRule}\titlerule}]
\titlespacing*{\chapter}{0pt}{-0.5em}{2.2em}
\titleformat{\section}
  {\normalfont\sffamily\Large\bfseries\color{BookForgeInk}}
  {\thesection}{0.7em}{}
\titlespacing*{\section}{0pt}{1.8em plus 0.2em minus 0.1em}{0.7em}
\titleformat{\subsection}
  {\normalfont\sffamily\large\bfseries\color{BookForgeInk}}
  {\thesubsection}{0.6em}{}
\titlespacing*{\subsection}{0pt}{1.3em plus 0.2em minus 0.1em}{0.45em}

\makeatletter
\@ifundefined{pandocbounded}{}{%
  \renewcommand*\pandocbounded[1]{%
    \sbox\pandoc@box{#1}%
    \Gscale@div\@tempa{\textheight}{\dimexpr\ht\pandoc@box+\dp\pandoc@box\relax}%
    \Gscale@div\@tempb{\linewidth}{\wd\pandoc@box}%
    \ifdim\@tempb\p@<\@tempa\p@\let\@tempa\@tempb\fi%
    \par\noindent\makebox[\linewidth][c]{%
      \ifdim\@tempa\p@<\p@\scalebox{\@tempa}{\usebox\pandoc@box}%
      \else\usebox{\pandoc@box}%
      \fi%
    }\par%
  }%
}
\makeatother
